## 개요

Day5 종합 실습(과제 3~4)의 **참고 정답 코드**입니다.  
ETL 파이프라인(Extract → Transform → Load → Report) 구조로 구현되어 있습니다.

### 파이프라인 흐름

```
create_spark_session() → extract_data() → transform_data() → load_data() → generate_report()
```

### 코드 구조

| 함수 | 역할 |
|------|------|
| `create_spark_session()` | Spark 클러스터 연결 |
| `extract_data()` | HDFS에서 CSV 읽기 |
| `transform_data()` | 4가지 분석 (카테고리/시간대/퍼널/유저행동) |
| `load_data()` | Parquet로 HDFS 저장 |
| `generate_report()` | 저장된 결과 읽어 리포트 출력 |
| `main()` | 전체 파이프라인 실행 및 예외 처리 |

## 실행 전 확인사항

- HDFS(namenode, datanode) 실행 중
- Spark(spark-master, pyspark-notebook) 실행 중

## 정답 코드

```python
from pyspark.sql import SparkSession
from pyspark.sql import functions as F
import logging
from datetime import datetime

# 로깅 설정
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


def create_spark_session():
    """SparkSession 생성"""
    return SparkSession.builder \
        .appName("E-commerce Analytics Pipeline") \
        .master("spark://spark-master:7077") \
        .getOrCreate()


def extract_data(spark, input_path):
    """HDFS에서 데이터 추출"""
    logger.info(f"📥 Extracting data from {input_path}")

    df = spark.read.csv(input_path, header=True, inferSchema=True)
    row_count = df.count()
    logger.info(f"✅ Extracted {row_count} records")

    return df


def transform_data(df):
    """데이터 변환 및 분석"""
    logger.info("🔄 Transforming data...")

    # 1. 타임스탬프 파싱 (CSV 문자열 → timestamp → hour 추출)
    df = df.withColumn("event_ts", F.to_timestamp("timestamp", "yyyy-MM-dd HH:mm:ss")) \
           .withColumn("hour", F.hour("event_ts"))

    # 2. 매출 계산 (purchase 이벤트만)
    df = df.withColumn(
        "revenue",
        F.when(F.col("event_type") == "purchase",
               F.col("price").cast("double") * F.col("quantity").cast("int")
               ).otherwise(F.lit(0.0))
    )

    # 3. 카테고리별 통계
    category_stats = df.groupBy("category").agg(
        F.count("*").alias("total_events"),
        F.countDistinct("user_id").alias("unique_users"),
        F.sum("revenue").alias("total_revenue"),
        F.sum(F.when(F.col("event_type") == "purchase", 1).otherwise(0)).alias("purchases")
    ).orderBy(F.col("total_revenue").desc())

    # 4. 시간대별 활동
    hourly_activity = df.groupBy("hour").agg(
        F.count("*").alias("events"),
        F.countDistinct("user_id").alias("active_users")
    ).orderBy("hour")

    # 5. 전환율 분석 (Funnel)
    funnel = df.groupBy("event_type").count().orderBy(F.col("count").desc())

    # 6. 사용자별 행동 (categories_viewed: view 이벤트의 카테고리만)
    user_behavior = df.groupBy("user_id").agg(
        F.count("*").alias("total_events"),
        F.countDistinct("session_id").alias("sessions"),
        F.round(F.sum("revenue"), 2).alias("total_spent"),
        F.sort_array(F.collect_set(
            F.when(F.col("event_type") == "view", F.col("category"))
        )).alias("categories_viewed")
    ).orderBy(F.col("total_spent").desc())

    logger.info("✅ Transformation completed")

    return {
        "category_stats": category_stats,
        "hourly_activity": hourly_activity,
        "funnel": funnel,
        "user_behavior": user_behavior
    }


def load_data(results, output_base_path):
    """처리 결과를 HDFS에 Parquet로 저장"""
    logger.info(f"💾 Loading data to {output_base_path}")

    for name, df in results.items():
        output_path = f"{output_base_path.rstrip('/')}/{name}"
        logger.info(f"Saving {name} to {output_path}")

        df.write.mode("overwrite").parquet(output_path)
        logger.info(f"✅ Saved {name}")


def generate_report(spark, output_base_path):
    """저장된 Parquet를 읽어 일일 리포트 출력"""
    logger.info("📊 Generating daily report...")

    base = output_base_path.rstrip("/")
    category_stats = spark.read.parquet(f"{base}/category_stats")
    hourly_activity = spark.read.parquet(f"{base}/hourly_activity")
    funnel = spark.read.parquet(f"{base}/funnel")
    user_behavior = spark.read.parquet(f"{base}/user_behavior")

    print("\n" + "=" * 60)
    print("E-COMMERCE DAILY REPORT - 2026-02-14")
    print("=" * 60)

    print("\n[1] Category Performance")
    category_stats.show(truncate=False)

    print("\n[2] Hourly Activity")
    hourly_activity.show(truncate=False)

    print("\n[3] Event Funnel")
    funnel.show(truncate=False)

    print("\n[4] Top 5 Users by Revenue")
    user_behavior.show(5, truncate=False)

    print("\n" + "=" * 60)
    logger.info("✅ Report generation completed")


def main():
    """메인 파이프라인 실행"""
    start_time = datetime.now()
    logger.info(f"🚀 Pipeline started at {start_time}")

    spark = None
    try:
        spark = create_spark_session()
        spark.sparkContext.setLogLevel("WARN")

        input_path = "hdfs://namenode:8020/user/data/raw/ecommerce/2026/02/14/ecommerce_logs.csv"
        output_path = "hdfs://namenode:8020/user/data/processed/ecommerce/2026/02/14"

        raw_data = extract_data(spark, input_path)
        results = transform_data(raw_data)
        load_data(results, output_path)
        generate_report(spark, output_path)

        duration = (datetime.now() - start_time).total_seconds()
        logger.info(f"🎉 Pipeline completed successfully in {duration:.2f} seconds")

    except Exception as e:
        logger.error(f"❌ Pipeline failed: {e}")
        import traceback
        traceback.print_exc()
        raise
    finally:
        if spark is not None:
            spark.stop()


if __name__ == "__main__":
    main()
```
